package practiceproblems;import java.util.Calendar;import java.util.Date;/** * * @author CEOCarlAllen */public class PracticeProblems {       public static void main(String[] args) {                   Date date = new Date();      Calendar calendar = Calendar.getInstance();           int day =  calendar.get(calendar.DAY_OF_WEEK);            System.out.println(Weekdays(day));      System.out.println(date);                      }                  public static String Weekdays(int Weekdays){          switch (Weekdays){              case 1: return "Sunday";               case 2: return "Monday";              case 3: return "Tuesday";              case 4: return "Wednesday";              case 5: return "Thrusday";              case 6: return "Friday";              case 7: return "Saturday";              default: return null;                        }      }         }
